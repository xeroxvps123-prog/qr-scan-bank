function doPost(e) {
  var sheet = SpreadsheetApp
    .getActiveSpreadsheet().getActiveSheet();
  var payload = JSON.parse(e.postData.contents);
  var values = sheet.getDataRange().getValues();
  var updated = 0;
  for (var i = 1; i < values.length; i++) {
    var empCode = String(values[i][1]).trim();
    for (var j = 0; j < payload.length; j++) {
      if (String(payload[j].empCode).trim()===empCode
          && payload[j].account) {
        sheet.getRange(i+1,5).setValue(payload[j].account);
        updated++; break;
      }
    }
  }
  return ContentService.createTextOutput(
    JSON.stringify({success:true,updated:updated})
  ).setMimeType(ContentService.MimeType.JSON);
}
function doGet(e){
  return ContentService.createTextOutput(
    JSON.stringify({status:"ok"})
  ).setMimeType(ContentService.MimeType.JSON);
}

📲 Mobile & HTTPS Help
To use the camera on your phone:
You need HTTPS. Here are 3 ways:
  • ngrok – quick HTTPS tunnel. Run ngrok http 8000 after starting a local server.
  • GitHub Pages – upload this HTML to a repository and enable Pages.
  • Netlify Drop – drag & drop the file to get a HTTPS URL instantly.
🔹 No camera? Use “Scan Image” – it works without HTTPS.
